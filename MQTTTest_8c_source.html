176  time_t now = time(NULL);

183

191  in->type >= 4 ? qm.tc[in->type - 4].name : "RAW", in->data.string.data);
192  msg_destroy(in);
193  free(in);
194  }
195  log_info(&state, 1, "Closing connections");
196  mqtt_closeConnection(mc);
197  mqtt_destroy_queue_map(&qm);
198  free(hMask);
199  free(host);
200  free(sysid);
201  return 0;
