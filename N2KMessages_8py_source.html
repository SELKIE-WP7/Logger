40 def pgnclass(number):
41  """! Decorator: Registers a class to parse the specified PGN
42 
43  Each class is required to have a static fromMessage method that parses an
44  ACTN2KMessage and returns a new instance of the class, or returns None on
45  failure.
46  """
47 
48  def pgnconverter(cl):
49  if number in _PGNRegistry:
50  raise KeyError(f"Converter already registered for PGN {number}")
51  _PGNRegistry[number] = cl.fromMessage
52  # Each PGN class must be a dataclass, so apply that here to save typing
53  # it above every single definition below
54  return dataclass(cl)
55 
56  return pgnconverter

74 def Timeseries(msgsource, pgns=None):
75  """! Convert stream of N2K messages into a timeseries
76 
77  msgsource is expected to be a generator yielding ACTN2KMessage instances
78  Will return a pandas DataFrame indexed by message timestamp (which are
79  assumed to be trusted)
80  """
81 
82  if pgns is None:
83  exclude = [60928]
84  pgns = [x for x in SupportedPGNs() if not x in exclude]
85 
86  from pandas import to_datetime, DataFrame as df
87 
88  skipped = False
89  messages = {}
90  cgroup = 0
91  ctimestamp = None
92  for msg in msgsource:
93  # Break after two consecutive failures to read messages
94  if msg is None:
95  if skipped:
96  break
97  skipped = True
98  continue
99  skipped = False
100  if msg.PGN not in pgns:
101  continue
102  cm = ConvertN2KMessage(msg)
103  if cm:
104  row = {k: v for k, v in zip(cm.fieldnames(), cm.fieldvalues())}
105  row["_N2KTS"] = msg.timestamp
106  if msg.timestamp == ctimestamp:
107  messages[cgroup].update(row)
108  else:
109  cgroup += 1
110  ctimestamp = msg.timestamp
111  messages[cgroup] = row
112  dat = df.from_dict(messages, orient="index").sort_index()
113  del messages
114  timecols = [x for x in dat.columns if x.endswith("Timestamp")]
115  for t in timecols:
116  dat[t] = to_datetime(dat[t], unit="s")
117  return dat

120 @dataclass
121 class ACTN2KMessage:
122  priority: int = 0
123  dst: int = 0
124  src: int = 0
125  timestamp: int = 0
126  length: int = 0
127  PGN: int = 0
128  datalen: int = 0
129  data: bytearray = field(default_factory=bytearray)
130  checksum: int = 0

132  @staticmethod
133  def findStart(data):
134  try:
135  return data.index(bytearray([ACTChar.ESC, ACTChar.SOT, ACTChar.N2K]))
136  except ValueError:
137  return None
138 
139  def findEnd(data):
140  try:
141  return data.index(bytearray([ACTChar.ESC, ACTChar.EOT]))
142  except ValueError:
143  return None

145  @staticmethod
146  def unpack(data):
147  spos = ACTN2KMessage.findStart(data)
148  if spos is None:
149  logging.debug("No message found")
150  return (None, 0)
151 
152  if (len(data) - spos) < 18:
153  # Insufficient data available
154  logging.debug("Buffer/data too small for any message")
155  return (None, 0)
156 
157  data = data[spos:]
158  if len(data) < data[3]:
159  logging.debug("Buffer/data too small for claimed message size")
160  # Insufficient data available
161  return (None, 0)
162 
163  msg = ACTN2KMessage()
164  msg.length = data[3]
165  msg.priority = data[4]
166  msg.PGN = data[5] + (data[6] * (2**8)) + (data[7] * (2**16))
167  msg.dst = data[8]
168  msg.src = data[9]
169  msg.timestamp = (
170  data[10]
171  + (data[11] * (2**8))
172  + (data[12] * (2**16))
173  + (data[13] * (2**24))
174  )
175  msg.datalen = data[14]
176 
177  datapos = 15
178  remaining = msg.datalen
179  while (len(data) - datapos) > 3 and remaining > 0:
180  c = data[datapos]
181  datapos += 1
182  remaining -= 1
183  if c == ACTChar.ESC:
184  nc = data[datapos]
185  datapos += 1
186  if nc == ACTChar.ESC:
187  msg.data.append(ACTChar.ESC)
188  continue
189  elif nc == ACTChar.EOT:
190  logging.debug("Message terminated early")
191  return (None, spos + datapos)
192  elif nc == ACTChar.SOT:
193  logging.debug("Unexpected message restart")
194  return (None, spos + datapos - 2)
195  else:
196  logging.debug(
197  f"Invalid escape sequence found in input data (ESC + 0x{nc:02x})"
198  )
199  return (None, spos + datapos)
200  else:
201  msg.data.append(c)
202 
203  if remaining:
204  logging.debug("Failed to read whole message?")
205 
206  msg.checksum = data[datapos]
207  datapos += 1
208 
209  if not (ACTN2KMessage.findEnd(data) == datapos):
210  logging.debug("End of message not in expected location")
211  else:
212  # Skip end of message bytes
213  datapos += 2
214 
215  return (msg, spos + datapos)

217  @staticmethod
218  def yieldFromFile(file):
219  x = file.read(1024)
220  msg = ACTN2KMessage.unpack(x)
221  x = x[msg[1] :]
222  while x:
223  yield msg[0]
224  msg = ACTN2KMessage.unpack(x)
225  x = x[msg[1] :]
226  if len(x) < 100 or msg[1] == 0:
227  d = file.read(1024 - len(x))
228  if d:
229  x = x + d
230  else:
231  break
232  # Out of data, so process until unable to parse further
233  while msg[1] > 0:
234  yield msg[0]
235  msg = ACTN2KMessage.unpack(x)
236  x = x[msg[1] :]
237 
238  return msg[0]

240  def calcChecksum(self):
241  csum = ACTChar.N2K
242  csum += self.length
243  csum += self.priority
244  for b in pack("<I", self.PGN)[0:3]:
245  csum += b
246  csum += self.dst
247  csum += self.src
248  for b in pack("<I", self.timestamp):
249  csum += b
250  csum += self.datalen
251  for b in self.data:
252  csum += b
253  return (256 - csum) % 256

279  def getFloat(self, offset, size, signed=True):
280  if size not in [8, 16, 32, 64]:
281  return ValueError("Unsupported size requested")
282  value = 0
283  m = "get"
284  if signed:
285  m += "Int"
286  else:
287  m += "UInt"
288  m += str(size)
289  value = getattr(self, m)(offset)
290  if (
291  ((not signed) and value == ((2**size) - 1))
292  or (signed and (value == -(2 ** (size - 1))))
293  or (signed and (value == (2 ** (size))))
294  ):
295  return float("NaN")
296 
297  return float(value)
