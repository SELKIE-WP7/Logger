70 def classify_messages(file):
71  """!
72  Read file and then classify/count messsages
73  @param file File to be processed
74  @returns ProcessedRecords() instance
75  """
76  unpacker = msgpack.Unpacker(file, unicode_errors="ignore")
77  out = SLMessageSink(msglogger=log)
78 
79  tots = ProcessedRecords()
80 
81  # Dict of Dicts gpsSeen[class][msg] = count and similar for NMEA
82  for msg in unpacker:
83  msg = out.Process(msg, output="raw")
84  if msg is None:
85  continue
86  tots.processed["MP"] += 1
87  if msg.SourceID in tots.classes["MP"]:
88  if msg.ChannelID in tots.classes["MP"][msg.SourceID]:
89  tots.classes["MP"][msg.SourceID][msg.ChannelID] += 1
90  else:
91  tots.classes["MP"][msg.SourceID][msg.ChannelID] = 1
92  else:
93  tots.classes["MP"][msg.SourceID] = {}
94  tots.classes["MP"][msg.SourceID][msg.ChannelID] = 1
95 
96  if msg.SourceID >= 0x10 and msg.SourceID < 0x20:
97  # Assume GPS
98  if msg.ChannelID != 0x03:
99  # We're only looking at raw messages on Ch 3
100  if not msg.ChannelID in [0, 1, 2]:
101  # But we will count messages that have been converted to other types
102  tots.processed["GPS"] = tots.processed["GPS"] + 1
103  continue
104  header = int.from_bytes(msg.Data[0:2], byteorder="big", signed=False)
105  ubxClass = msg.Data[2]
106  ubxMsg = msg.Data[3]
107 
108  if ubxClass in tots.classes["GPS"]:
109  if ubxMsg in tots.classes["GPS"][ubxClass]:
110  tots.classes["GPS"][ubxClass][ubxMsg] = (
111  tots.classes["GPS"][ubxClass][ubxMsg] + 1
112  )
113  else:
114  tots.classes["GPS"][ubxClass][ubxMsg] = 1
115  else:
116  tots.classes["GPS"][ubxClass] = {}
117  tots.classes["GPS"][ubxClass][ubxMsg] = 1
118 
119  elif msg.SourceID >= 0x30 and msg.SourceID < 0x40:
120  # Assume NMEA
121  if msg.ChannelID != 0x03:
122  if not msg.ChannelID in [0, 1, 2]:
123  tots.processed["NMEA"] = tots.processed["NMEA"] + 1
124  continue
125  nmeaHead = msg.Data[0:10].decode("ascii", "replace")
126  if nmeaHead[1] == "P":
127  talker = nmeaHead[1:5]
128  message = nmeaHead[5:8]
129  else:
130  talker = nmeaHead[1:3]
131  message = nmeaHead[3:6]
132 
133  if talker in tots.classes["NMEA"]:
134  if message in tots.classes["NMEA"][talker]:
135  tots.classes["NMEA"][talker][message] = (
136  tots.classes["NMEA"][talker][message] + 1
137  )
138  else:
139  tots.classes["NMEA"][talker][message] = 1
140  else:
141  tots.classes["NMEA"][talker] = {}
142  tots.classes["NMEA"][talker][message] = 1
143 
144  tots.SourceMap = out.SourceMap()
145  return tots
