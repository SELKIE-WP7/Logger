24 def log_to_file(port, output, baud=115200, timeout=1):
25  s = serial.Serial(port, baud, timeout=timeout)
26 
27  with open(output, "wb") as out:
28  print("Press CTRL+C to exit")
29  try:
30  while True:
31  data = s.read(s.in_waiting or 1)
32  if data:
33  out.write(data)
34  except KeyboardInterrupt:
35  pass

38 def SLSerialLog():
39  import argparse
40 
41  options = argparse.ArgumentParser(
42  description="Dump serial data to file",
43  epilog="Created as part of the SELKIE project",
44  )
45  options.add_argument(
46  "-o", "--output", metavar="FILE", help="Output file name", default="serial.dat"
47  )
48  options.add_argument(
49  "-p", "--port", metavar="PORT", help="Serial port name", default=None
50  )
51  options.add_argument(
52  "-b", "--baud", metavar="BAUD", help="Serial port data rate", default=115200
53  )
54  options.add_argument(
55  "-t",
56  "--timeout",
57  metavar="SECONDS",
58  help="Serial port read timeout, in seconds",
59  default=1,
60  )
61 
62  args = options.parse_args()
63 
64  if args.port is None:
65  port = ask_for_port()
66  else:
67  port = args.port
68 
69  return log_to_file(port, args.output, args.baud, args.timeout)
