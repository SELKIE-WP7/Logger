65 def getSourceMap(filename, progress=None, stop=None):
66  file = open(filename, "rb")
67  file.seek(0, os.SEEK_END)
68  size = file.tell()
69  file.seek(0, os.SEEK_SET)
70  unpacker = msgpack.Unpacker(file, unicode_errors="ignore")
71  out = SLMessageSink(msglogger=logging.getLogger("Messages"))
72 
73  stats = {}
74  for msg in unpacker:
75  if stop and stop():
76  return None
77  msg = out.Process(msg, output="raw")
78  if msg is None:
79  continue
80 
81  if not msg.SourceID in stats:
82  stats[msg.SourceID] = {}
83  if not msg.ChannelID in stats[msg.SourceID]:
84  stats[msg.SourceID][msg.ChannelID] = 1
85  else:
86  stats[msg.SourceID][msg.ChannelID] += 1
87 
88  if progress:
89  progress(file.tell() / size)
90  return (out.SourceMap(), stats)
