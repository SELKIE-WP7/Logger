25  """!
26  Configure custom logging instance, using the functions from the rich
27  package if it is available and falling back to standard definitions
28  otherwise.
29  @returns Custom logger object, with some module items added for convenience.
30  """
31  log = logging.getLogger("")
32  try:
33  from rich.logging import RichHandler
34 
35  _rh = RichHandler(enable_link_path=False)
36  log.addHandler(_rh)
37 
38  except ImportError:
39  # 'rich' package not available, so fall back to standard config
40  _lh = logging.StreamHandler()
41  # Set the handler to lowest (standard) level
42  # Actual output will be filtered based on the level set for log (WARNING by default)
43  _lh.setLevel(logging.DEBUG)
44  _lf = logging.Formatter("[%(asctime)s]\t%(levelname)s\t%(message)s")
45  _lh.setFormatter(_lf)
46  log.addHandler(_lh)
47 
48  # Expose some functions and constants from the logging module
49  log.DEBUG = logging.DEBUG
50  log.INFO = logging.INFO
51  log.WARNING = logging.WARNING
52  log.ERROR = logging.ERROR
53  log.CRITICAL = logging.CRITICAL
54  log.getLevelName = logging.getLevelName
